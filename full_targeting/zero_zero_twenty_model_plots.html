def get_payoff(A, B, R, x_0, T, delta): 
    Q = (1/3) * np.identity(3)
    K_t = Q
    K_sequence = [Q]
    for i in range(T):
        K_t = delta * A.T @ (K_t - K_t @ B @ np.linalg.inv(B.T @ K_t @ B + R / delta) @ B.T @ K_t) @ A + Q
        K_sequence.insert(0, K_t)

    x_t = x_0
    r_ts = []
    payoff = 0
    payoffs = []
    x_ts = [x_0]
    i = 0
    for K in K_sequence[1:]:
        L = -(np.linalg.inv(B.T @ K @ B + R / delta) @ B.T @ K @ A)
        r_t = L @ x_t
        r_ts.append(r_t.item())
        payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item() + (-1 * delta**i * (R * r_t**2)).item()
        payoffs.append(payoff)
        x_t_new = A @ x_t + B @ r_t
        x_ts.append(x_t_new)
        x_t = x_t_new
        i += 1
        
    # THE TERMINAL PAYOFF TERM AT TIME T:
    payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item()
    payoffs.append(payoff)
        
    return payoff, payoffs

def solve(target, R, x_0, T, delta, a11):
    b = 0.1 # the influence weight
    # NEW A MATRIX GOES HERE:
    A = np.array([
      [a * [1.0, 1.0-b][target == 1] for a in [a11, (1 - a11)/2, (1 - a11)/2]],
      [a * [1.0, 1.0-b][target == 2] for a in [1/3, 1/3, 1/3]],
      [a * [1.0, 1.0-b][target == 3] for a in [1/3, 1/3, 1/3]],
    ], ndmin = 2)
    B = np.array([
      [[0.0, b][target == 1]],
      [[0.0, b][target == 2]],
      [[0.0, b][target == 3]],
    ], ndmin = 2)
    return get_payoff(A, B, R, x_0, T, delta)

x_0 = np.array([[0], [0], [20]], ndmin = 2)
T = 100
delta = 1
R = 0
a11 = 0.8
for agent in [1, 2, 3]:
    payoff, payoffs = solve(agent, R, x_0, T, delta, a11)
    print(payoff, "payoff for targeting agent", agent)
    plt.plot(range(T+1), payoffs, label = f"target agent {agent}")
plt.legend()
plt.xlabel("time")
plt.ylabel("cumulative payoff")
plt.title(f"a11 = {a11}")
plt.show()
