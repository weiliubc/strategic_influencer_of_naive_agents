def get_payoff(A, B, R, x_0, T, delta): 
    n = x_0.size
    Q = (1/n) * np.identity(n)
    K_t = Q
    K_sequence = [Q]
    for i in range(T):
        K_t = delta * A.T @ (K_t - K_t @ B @ np.linalg.inv(B.T @ K_t @ B + R / delta) @ B.T @ K_t) @ A + Q
        K_sequence.insert(0, K_t)

    x_t = x_0
    r_ts = []
    payoff = 0
    payoffs = []
    x_ts = [x_0]
    i = 0
    for K in K_sequence[1:]:
        L = -(np.linalg.inv(B.T @ K @ B + R / delta) @ B.T @ K @ A)
        r_t = L @ x_t
        r_ts.append(r_t.item())
        payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item() + (-1 * delta**i * (R * r_t**2)).item()
        payoffs.append(payoff)
        x_t_new = A @ x_t + B @ r_t
        x_ts.append(x_t_new)
        x_t = x_t_new
        i += 1
        
    return payoff, payoffs, r_ts, K_sequence, x_ts

def solveit(A, B, x_0 = np.array([[4], [4]], ndmin = 2)):
    T = 30
    R = 0
    delta = 1
    return get_payoff(A, B, R, x_0, T, delta)

  [0.6 * 0.99, 0.4 * 0.99],
  [0.6, 0.4]
], ndmin = 2)
B = np.array([
  [0.01], 
  [0.0]
], ndmin = 2)
_, payoffs, _, _, _ = solveit(A, B)

A2 = np.array([
  [0.6, 0.4],
  [0.6 * 0.99, 0.4 * 0.99]
], ndmin = 2)
B2 = np.array([
  [0.0], 
  [0.01]
], ndmin = 2)
_, payoffs2, _, _, _ = solveit(A2, B2)

plt.plot(range(30), payoffs, label = "target agent 1")
plt.plot(range(30), payoffs2, label = "target agent 2")
plt.xlabel("time")
plt.ylabel("cumulative payoff")
plt.legend()
plt.title("Initial Opinions: (4, 4)")
plt.show()

Targeting the first agent yields the better payoff when the initial opinions are identical (4, 4).

$\displaystyle \left[\begin{matrix}\frac{K_{11} + a_{1}^{2} \left(K_{11} K_{22} - K_{12}^{2}\right)}{K_{11}} & \frac{a_{1} a_{2} \left(K_{11} K_{22} - K_{12}^{2}\right)}{K_{11}}\\\frac{a_{1} a_{2} \left(K_{11} K_{22} - K_{12}^{2}\right)}{K_{11}} & \frac{K_{11} + a_{2}^{2} \left(K_{11} K_{22} - K_{12}^{2}\right)}{K_{11}}\end{matrix}\right]$

Note the $\frac{det(K)}{K_{11}}$ term is common to all entries and can be factored out ($det(K)$ is $K_{11}K_{22} - K_{12}^2$).

This becomes:

$$Q + \frac{det(K)}{K_{11}}\begin{bmatrix}a_1^2 & a_1a_2\\a_1a_2 & a_2^2\end{bmatrix}$$

which matches what was already known about the setup.
