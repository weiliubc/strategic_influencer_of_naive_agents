def get_payoff(A, B, R, x_0, T, delta): 
    Q = (1/x_0.size) * np.identity(x_0.size)
    K_t = Q
    K_sequence = [Q]
    for i in range(T):
        K_t = delta * A.T @ (K_t - K_t @ B @ np.linalg.inv(B.T @ K_t @ B + R / delta) @ B.T @ K_t) @ A + Q
        K_sequence.insert(0, K_t)

    x_t = x_0
    r_ts = []
    payoff = 0
    payoffs = []
    x_ts = [x_0]
    i = 0
    for K in K_sequence[1:]:
        L = -(np.linalg.inv(B.T @ K @ B + R / delta) @ B.T @ K @ A)
        r_t = L @ x_t
        r_ts.append(r_t.item())
        payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item() + (-1 * delta**i * (R * r_t**2)).item()
        payoffs.append(payoff)
        x_t_new = A @ x_t + B @ r_t
        x_ts.append(x_t_new)
        x_t = x_t_new
        i += 1
        
    return payoff, payoffs

def solve(target, delta = 1, x_0 = np.array([[-10], [7], [20]], ndmin = 2)):
    T = 30
    R = 0
    b = 0.1 # the influence weight
    A = np.array([
      [a * [1.0, 1.0-b][target == INFLUENTIAL] for a in A_nobody[0]], # influential agent
      [a * [1.0, 1.0-b][target == FOLLOWER] for a in A_nobody[1]], # follower agent
      [a * [1.0, 1.0-b][target == STUBBORN] for a in A_nobody[2]], # stubborn agent
    ], ndmin = 2)
    #print("A")
    #print(A) # print this out to confirm the right agent was selected

    B = np.array([
      [[0.0, b][target == INFLUENTIAL]],
      [[0.0, b][target == FOLLOWER]],
      [[0.0, b][target == STUBBORN]],
    ], ndmin = 2)
    #print()
    #print("B")
    #print(B) # also print this out
    #print()
    #print()

    return get_payoff(A, B, R, x_0, T, delta)
