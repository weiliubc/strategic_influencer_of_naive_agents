def get_payoff(A, B, R, x_0, T, delta): 
    Q = (1/3) * np.identity(3)
    K_t = Q
    K_sequence = [Q]
    for i in range(T):
        K_t = delta * A.T @ (K_t - K_t @ B @ np.linalg.inv(B.T @ K_t @ B + R / delta) @ B.T @ K_t) @ A + Q
        K_sequence.insert(0, K_t)

    x_t = x_0
    r_ts = []
    payoff = 0
    payoffs = []
    x_ts = [x_0]
    i = 0
    for K in K_sequence[1:]:
        L = -(np.linalg.inv(B.T @ K @ B + R / delta) @ B.T @ K @ A)
        r_t = L @ x_t
        r_ts.append(r_t.item())
        payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item() + (-1 * delta**i * (R * r_t**2)).item()
        payoffs.append(payoff)
        x_t_new = A @ x_t + B @ r_t
        x_ts.append(x_t_new)
        x_t = x_t_new
        i += 1
        
    return payoff, payoffs

def solve(target, R, x_0, T, delta):
    b = 0.1 # the influence weight
    # NEW A MATRIX GOES HERE:
    A = np.array([
      [a * [1.0, 1.0-b][target == INFLUENTIAL] for a in [0.6, 0.2, 0.2]], # influential agent
      [a * [1.0, 1.0-b][target == FOLLOWER]    for a in [0.5, 0.5, 0.0]], # follower agent
      [a * [1.0, 1.0-b][target == STUBBORN]    for a in [0.2, 0.0, 0.8]], # stubborn agent
    ], ndmin = 2)
    B = np.array([
      [[0.0, b][target == INFLUENTIAL]],
      [[0.0, b][target == FOLLOWER]],
      [[0.0, b][target == STUBBORN]],
    ], ndmin = 2)
    return get_payoff(A, B, R, x_0, T, delta)

x_0 = np.array([[-10], [7], [20]], ndmin = 2) # extreme agent 3 opinion
T = 100
delta = 1
R = 0
for agent in ["INFLUENTIAL", "FOLLOWER", "STUBBORN"]:
    payoff, payoffs = solve(globals()[agent], R, x_0, T, delta)
    print(payoff, "payoff for targeting", agent.lower(), "agent")
    plt.plot(range(T), payoffs, label = "target " + agent.lower() + " agent")
plt.legend()
plt.xlabel("time")
plt.ylabel("cumulative payoff")
plt.show()

-350.84373803636794 payoff for targeting influential agent
-450.0741735566636 payoff for targeting follower agent
-185.35781456709256 payoff for targeting stubborn agent

Optimal targeting order is exactly 1, 2, 3.
