def get_payoff(A, B, R, x_0, T, delta): 
    Q = (1/3) * np.identity(3)
    K_t = Q
    K_sequence = [Q]
    for i in range(T):
        K_t = delta * A.T @ (K_t - K_t @ B @ np.linalg.inv(B.T @ K_t @ B + R / delta) @ B.T @ K_t) @ A + Q
        K_sequence.insert(0, K_t)

    x_t = x_0
    r_ts = []
    payoff = 0
    payoffs = []
    x_ts = [x_0]
    i = 0
    for K in K_sequence[1:]:
        L = -(np.linalg.inv(B.T @ K @ B + R / delta) @ B.T @ K @ A)
        r_t = L @ x_t
        r_ts.append(r_t.item())
        payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item() + (-1 * delta**i * (R * r_t**2)).item()
        payoffs.append(payoff)
        x_t_new = A @ x_t + B @ r_t
        x_ts.append(x_t_new)
        x_t = x_t_new
        i += 1
        
    # THE TERMINAL PAYOFF TERM AT TIME T:
    payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item()
    payoffs.append(payoff)
        
    return payoff, payoffs, x_ts

def get_kss(A, B, R, x_0, T, delta): 
    Q = (1/3) * np.identity(3)
    K_t = Q
    K_sequence = [Q]
    while True:
        K_t_new = delta * A.T @ (K_t - K_t @ B @ np.linalg.inv(B.T @ K_t @ B + R / delta) @ B.T @ K_t) @ A + Q
        K_sequence.insert(0, K_t_new)
        if np.allclose(K_t, K_t_new, rtol = np.finfo(K_t.dtype).eps, atol = np.finfo(K_t.dtype).eps):
            return K_t
        K_t = K_t_new

def solve(target, x_0, base, T = 30, kss = False):
    delta = 1
    R = 0
    b = 0.01 # the influence weight
    A = np.array([
      [a * [1.0, 1.0-b][target == CENTRAL] for a in base[0]], # central
      [a * [1.0, 1.0-b][target == FOLLOW1] for a in base[1]], # peripheral 1
      [a * [1.0, 1.0-b][target == FOLLOW2] for a in base[2]], # peripheral 2
    ], ndmin = 2)
    B = np.array([
      [[0.0, b][target == CENTRAL]],
      [[0.0, b][target == FOLLOW1]],
      [[0.0, b][target == FOLLOW2]],
    ], ndmin = 2)
    operator = ["get_payoff", "get_kss"][kss]
    return globals()[operator](A, B, R, x_0, T, delta)

def generate_plot(x_0, base):
    for agent in ["CENTRAL", "FOLLOW1", "FOLLOW2"]:
        payoff, payoffs, _ = solve(globals()[agent], x_0, base)
        plt.plot(range(len(payoffs)), payoffs, label = "target " + agent.lower() + " agent")
    plt.legend()
    plt.xlabel("time")
    plt.ylabel("cumulative payoff")
    plt.title("x_0 = " + str(x_0.T))
    plt.show()

General payoff form is $-K_{11} + 6K_{12} - 9K_{22}$ given the 2x2 structure above.

Therefore payoff increases if $K_{12}$ increases or one of $K_{22}$, $K_{11}$ decrease.

A decrease in $K_{22}$ has more weight than an increase in $K_{12}$ or a decrease in $K_{11}$.

Similarly, an increase in $K_{12}$ has more weight than a decrease in $K_{11}$.

These weights come from the $x_0$ opinions.

In going from targeting the first follower to the second follower, there is a loss in the off-diagonal weight corresponding to the entry for $a_{13}$ and a smaller loss in the on-diagonal weight of agent 3. But because the on-diagonal change is weighed more due to the initial opinions, the dominating effect is the on-diagonal change.

Compared to targeting the second follower, there is an increase in the first term but a much more significant decrease in the other two terms, so this is less optimal.

The question now is whether there is a way to quantify the extent to which the weights caused by the opinions impact the $K_{ss}$, as well as quantify how $K_{ss}$ changes between two targeting cases.

We, in general, have the following form for the payoff where $x_0^2 = 0$:

$$-(x_0^1)^2 K_{11} - 2x_0^1 x_0^3 K_{13} -(x_0^3)^2 K_{33}$$
