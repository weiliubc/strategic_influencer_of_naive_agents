def get_payoff(A, B, R, x_0, T, delta): 
    Q = (1/3) * np.identity(3)
    K_t = Q
    K_sequence = [Q]
    for i in range(T):
        K_t = delta * A.T @ (K_t - K_t @ B @ np.linalg.inv(B.T @ K_t @ B + R / delta) @ B.T @ K_t) @ A + Q
        K_sequence.insert(0, K_t)

    x_t = x_0
    r_ts = []
    payoff = 0
    payoffs = []
    x_ts = [x_0]
    i = 0
    for K in K_sequence[1:]:
        L = -(np.linalg.inv(B.T @ K @ B + R / delta) @ B.T @ K @ A)
        r_t = L @ x_t
        r_ts.append(r_t.item())
        payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item() + (-1 * delta**i * (R * r_t**2)).item()
        payoffs.append(payoff)
        x_t_new = A @ x_t + B @ r_t
        x_ts.append(x_t_new)
        x_t = x_t_new
        i += 1
        
    # THE TERMINAL PAYOFF TERM AT TIME T:
    payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item()
    payoffs.append(payoff)
        
    return payoff, payoffs, x_ts, r_ts

def solve(target, x_0, base, delta, T, R):
    b = 0.01 # the influence weight
    A = np.array([
      [a * [1.0, 1.0-b][target == CENTRAL] for a in base[0]], # central
      [a * [1.0, 1.0-b][target == FOLLOW1] for a in base[1]], # peripheral 1
      [a * [1.0, 1.0-b][target == FOLLOW2] for a in base[2]], # peripheral 2
    ], ndmin = 2)
    B = np.array([
      [[0.0, b][target == CENTRAL]],
      [[0.0, b][target == FOLLOW1]],
      [[0.0, b][target == FOLLOW2]],
    ], ndmin = 2)
    return get_payoff(A, B, R, x_0, T, delta)

def generate_plot(name, x_0, base, delta = 1, T = 30, R = 0, expected_overlap = False, do_not_print = False):
    labels = {
        "CENTRAL": "(1) target central agent",
        "FOLLOW1": "(2) target 1st peripheral agent",
        "FOLLOW2": "(3) target 2nd peripheral agent"
    }
    for i, agent in enumerate(["CENTRAL", "FOLLOW1", "FOLLOW2"]):
        payoff, payoffs, x_t, r_t = solve(globals()[agent], x_0, base, delta, T, R)
        if not do_not_print:
            print(f"Messages for targeting agent {i+1} ({labels[agent]}):")
            print(r_t)
            print(f"Opinions for targeting agent {i+1} ({labels[agent]}):")
            sum_payoff = 0
            for x in x_t:
                unit = x.T @ ((1/3)*np.identity(3)) @ x
                sum_payoff += unit
                print(x.T, "total cost =", sum_payoff.item(), "stage cost =", unit.item())
            print()
        if expected_overlap and agent != "CENTRAL":
            if agent == "FOLLOW1":
                plt.plot(range(len(payoffs)), payoffs, label = labels[agent], linewidth = 4, color = "orange")
            elif agent == "FOLLOW2":
                plt.plot(range(len(payoffs)), payoffs, label = labels[agent], color = "green")
        else:
            plt.plot(range(len(payoffs)), payoffs, label = labels[agent])
    plt.legend()
    plt.xlabel("time")
    plt.ylabel("cumulative payoff")
    plt.title(fr"x_0 = {x_0.T[0]}, $\delta$ = {delta}")
    if name != None:
        plt.savefig(f"#plots/{name}.png", format = "png", dpi = 300, facecolor= (1, 1, 1))
    plt.show()

def steady_K(target, base, n, delta = 1):
    R = 0
    b = 0.01 # the influence weight
    A = np.array([
          [a * [1.0, 1.0-b][target == CENTRAL] for a in base[0]], # central
          [a * [1.0, 1.0-b][target == FOLLOW1] for a in base[1]], # peripheral 1
          [a * [1.0, 1.0-b][target == FOLLOW2] for a in base[2]], # peripheral 2
        ], ndmin = 2)
    B = np.array([
      [[0.0, b][target == CENTRAL]],
      [[0.0, b][target == FOLLOW1]],
      [[0.0, b][target == FOLLOW2]],
    ], ndmin = 2)
    Q = (1/n) * np.identity(n)
    K_t = Q
    K_sequence = [Q]
    while True:
        K_t_new = delta * A.T @ (K_t - K_t @ B @ np.linalg.inv(B.T @ K_t @ B + R / delta) @ B.T @ K_t) @ A + Q
        K_sequence.insert(0, K_t_new)
        if np.allclose(K_t, K_t_new, rtol = np.finfo(K_t.dtype).eps, atol = np.finfo(K_t.dtype).eps):
            return K_t_new
        K_t = K_t_new
