def payoff(aii, aij, n, x_0, R, delta, T = 20):
    A = (aii - aij) * np.identity(n) + aij * np.ones((n, n))
    b = 1 - (aii + (n - 1) * aij) # b is produced using aii, aij, n
    B = b * np.ones((n, 1))

    Q = (1/n) * np.identity(n)
    K_t = Q
    K_sequence = [Q]
    for i in range(T):
        K_t = delta * A.T @ (K_t - K_t @ B @ np.linalg.inv(B.T @ K_t @ B + R / delta) @ B.T @ K_t) @ A + Q
        K_sequence.insert(0, K_t)

    x_t = x_0
    r_ts = []
    payoff = 0
    xqx = 0
    rrr = 0
    x_ts = [x_0]
    i = 0
    for K in K_sequence[1:]:
        L = -(np.linalg.inv(B.T @ K @ B + R / delta) @ B.T @ K @ A)
        r_t = L @ x_t
        r_ts.append(r_t.item())
        payoff += (-1 * delta**i * (x_t.T @ Q @ x_t)).item() + (-1 * delta**i * (R * r_t**2)).item()
        xqx += (-1 * delta**i * (x_t.T @ Q @ x_t)).item()
        rrr += (-1 * delta**i * (R * r_t**2)).item()
        x_t_new = A @ x_t + B @ r_t
        x_ts.append(x_t_new)
        x_t = x_t_new
        i += 1
        
    return payoff, xqx, rrr

def expr(R, index, name, lbound = 0, ubound = 0.5): # general experiment function
    x_0 = np.array([[5.0], [10.0]], ndmin = 2)
    grid = np.linspace(lbound, ubound, 100) # b from very small to just under 0.5
    results = []
    for b in grid:
        results.append(payoff(1 - b, 0, 2, x_0, R, 1)[index]) # delta = 1, n = 2

    plt.plot(grid, results)
    plt.title(f"{name} vs b: R = {R}")
    plt.xlabel("b")
    plt.ylabel(name)
    plt.show()
    
def multiplot(R, lbound = 0, ubound = 0.5): # general experiment function
    x_0 = np.array([[5.0], [10.0]], ndmin = 2)
    grid = np.linspace(lbound, ubound, 100) # b from very small to just under 0.5
    for index in range(3):
        results = []
        for b in grid:
            results.append(payoff(1 - b, 0, 2, x_0, R, 1)[index]) # delta = 1, n = 2
        plt.plot(grid, results, label = ["total payoff", "x'Qx term", "r'Rr term"][index])
        
    plt.title(f"payoffs vs b: R = {R}")
    plt.xlabel("b")
    plt.ylabel("payoffs")
    plt.legend()
    plt.show()
