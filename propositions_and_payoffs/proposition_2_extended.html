def eigenconverge(aii, aij, n, x_0, R, delta, eigenplot = False):
    A = (aii - aij) * np.identity(n) + aij * np.ones((n, n))
    b = 1 - (aii + (n - 1) * aij)
    bl = b * (-b * (1 - b) * (1 / n) / (b**2 + R / delta))
    BL = bl * np.ones((n, n))
    ABL = A + BL
    #print((1 - b) * (R / delta) / (b**2 + R / delta))
    
    vals, vecs = np.linalg.eig(A)
    c_vec = np.linalg.inv(vecs) @ x_0
    t = 0
    x_t_simple = [x_0]
    x_t_testsimple = [x_0]
    while True:
        t += 1
        x_t = np.linalg.matrix_power(A, t) @ x_0
        x_t_test = vecs @ ((np.array(vals, ndmin = 2).T ** t) * c_vec)
        if not np.allclose(x_t, x_t_test):
            raise Exception("failure in simple")
        x_t_simple.append(x_t)
        x_t_testsimple.append(x_t_test)
        if np.allclose(x_t, np.zeros(n)):
            break

    if not eigenplot: plt.plot(range(t+1), [np.max(x) for x in x_t_simple], label = "Max Opinion, Simple Strategy")
    else: plt.plot(range(t+1), [np.max(x) for x in x_t_testsimple], label = "Max Opinion, Simple Strategy")
    
    vals, vecs = np.linalg.eig(ABL)
    c_vec = np.linalg.inv(vecs) @ x_0
    t = 0
    x_t_myopic = [x_0]
    x_t_testmyopic = [x_0]
    while True:
        t += 1
        x_t = np.linalg.matrix_power(ABL, t) @ x_0
        x_t_test = vecs @ ((np.array(vals, ndmin = 2).T ** t) * c_vec)
        if not np.allclose(x_t, x_t_test):
            raise Exception("failure in myopic")
        x_t_myopic.append(x_t)
        x_t_testmyopic.append(x_t_test)
        if np.allclose(x_t, np.zeros(n)):
            break

    if not eigenplot: plt.plot(range(t+1), [np.max(x) for x in x_t_myopic], label = "Max Opinion, Myopic Strategy")
    else: plt.plot(range(t+1), [np.max(x) for x in x_t_testmyopic], label = "Max Opinion, Myopic Strategy")
    plt.plot([], [], color = "white", label = "$a_{ii} = "+str(aii)+", a_{ij} = "+str(aij)+f", n = {n}, R = {R},"+" \delta = "+str(delta)+"$")
    plt.plot([], [], color = "white", label = "$x_0 = "+str(x_0.T)+"$")
    if not eigenplot: plt.title("Convergence of Opinions using Simple and Myopic Strategies: Matrix Power Method")
    else: plt.title("Convergence of Opinions using Simple and Myopic Strategies: Eigenvalue Power Method")
    plt.legend()
    plt.xlabel("Time")
    plt.ylabel("Maximum Naive Agent Opinion")
    plt.show()
    
x_0 = np.array([[-5.0], [-5.0], [-10.0]], ndmin = 2)
eigenconverge(0.4, 0.2, 3, x_0, 0.2, 0.8)

def largest_eigenvalue(aii, aij, n, x_0, R, delta):
    A = (aii - aij) * np.identity(n) + aij * np.ones((n, n))
    b = 1 - (aii + (n - 1) * aij)
    bl = b * (-b * (1 - b) * (1 / n) / (b**2 + R / delta))
    BL = bl * np.ones((n, n))
    ABL = A + BL
    #print((1 - b) * (R / delta) / (b**2 + R / delta))
    
    vals, vecs = np.linalg.eig(A)
    c_vec = np.linalg.inv(vecs) @ x_0
    t = 0
    x_t_simple = [np.max(np.abs(vals))]
    while True:
        t += 1
        x_t = np.linalg.matrix_power(A, t) @ x_0
        x_t_test = vecs @ ((np.array(vals, ndmin = 2).T ** t) * c_vec)
        if not np.allclose(x_t, x_t_test):
            raise Exception("failure in simple")
        x_t_simple.append(np.max(np.abs(np.array(vals, ndmin = 2).T ** t)))
        if np.allclose(x_t, np.zeros(n)):
            break

    vals, vecs = np.linalg.eig(ABL)
    c_vec = np.linalg.inv(vecs) @ x_0
    t = 0
    x_t_myopic = [np.max(np.abs(vals))]
    while True:
        t += 1
        x_t = np.linalg.matrix_power(ABL, t) @ x_0
        x_t_test = vecs @ ((np.array(vals, ndmin = 2).T ** t) * c_vec)
        if not np.allclose(x_t, x_t_test):
            raise Exception("failure in myopic")

        x_t_myopic.append(np.max(np.abs(np.array(vals, ndmin = 2).T ** t)))
        if np.allclose(x_t, np.zeros(n)):
            break

    plt.plot(range(len(x_t_simple)), x_t_simple, label = "Simple Strategy")
    plt.plot(range(len(x_t_myopic)), x_t_myopic, label = "Myopic Strategy")
    plt.plot([], [], color = "white", label = "$a_{ii} = 0.4, a_{ij} = 0.2, n = 3, R = 0.2, \delta = 0.8$")
    plt.plot([], [], color = "white", label = "$x_0 = [-5, -5, -10]$")
    plt.title("Max Observed Network Eigenvalues using Simple and Myopic Strategies")
    plt.legend()
    plt.xlabel("Time")
    plt.ylabel("Maximum Eigenvalue (absolute value)")
    plt.show()
